&lt;style&gt; /*计算器层样式*/ .caculator{ width:300px;/*宽度*/ border:1px solid darkred;/*边框（边框宽度 边框样式 边框颜色）*/ } /*文本框样式*/ .result{ width:296px;/*宽度*/ height: 50px;/*高度*/ font-size: 30px;/*字体大小*/ text-align: right;/*文本对齐方式*/ } /*按钮层ul样式*/ .btns{ margin:0;/*外边距*/ padding:0;/*内边距*/ list-style: none;/*li列表样式*/ margin-left:3px;/*左外边距*/ } /*按钮层li样式*/ .btns li{ width:70px;/*宽度*/ height:70px;/*高度*/ line-height: 70px;/*行高(一般用来设置垂直居中)*/ border: 1px solid darkseagreen;/*边框（边框宽度 边框样式 边框颜色）*/ border-radius: 5px;/*边框圆角*/ float: left;/*浮动*/ margin-right:2px;/*右外边距*/ margin-bottom: 2px;/*下外边距*/ text-align: center;/*文本水平对齐方式*/ background: lavenderblush;/*背景颜色*/ } /*按钮层li鼠标经过的样式*/ .btns li:hover{ background: deeppink;/*背景颜色*/ color: white;/*字体颜色*/ } &lt;/style&gt; &lt;/head&gt; &lt;body&gt; &lt;div class="caculator"&gt; &lt;input type="text" class="result" id="result" readonly="readonly"&gt; &lt;ul class="btns" id="btnsUL"&gt; &lt;li class="num"&gt;1&lt;/li&gt;&lt;li class="num"&gt;2&lt;/li&gt;&lt;li class="num"&gt;3&lt;/li&gt;&lt;li class="operator"&gt;+&lt;/li&gt; &lt;li class="num"&gt;4&lt;/li&gt;&lt;li class="num"&gt;5&lt;/li&gt;&lt;li class="num"&gt;6&lt;/li&gt;&lt;li class="operator"&gt;-&lt;/li&gt; &lt;li class="num"&gt;7&lt;/li&gt;&lt;li class="num"&gt;8&lt;/li&gt;&lt;li class="num"&gt;9&lt;/li&gt;&lt;li class="operator"&gt;*&lt;/li&gt; &lt;li&gt;C&lt;/li&gt;&lt;li class="num"&gt;0&lt;/li&gt;&lt;li&gt;=&lt;/li&gt;&lt;li class="operator"&gt;/&lt;/li&gt; &lt;/ul&gt; &lt;!-- 清除浮动 --&gt; &lt;div style="clear: both;"&gt;&lt;/div&gt; &lt;/div&gt; &lt;script&gt; var ulDom = document.getElementById("btnsUL");//获取ul对象 var liDoms = ulDom.getElementsByTagName("li");//获取ul里的li元素对象 var resultDom = document.getElementById("result");//获取result文本框对象

var num=null;//数字变量，用来保存第一次点击的数字
var operator=null;//运算符变量，用来保存点击的运算

//利用遍历给所有的li元素绑定事件
for(var i=0;i<liDoms.length;i++){
//给li对象绑定单击事件
liDoms[i].onclick = liHandler;
}

//li元素单击事件的处理函数
function liHandler(oEvent){
//获取事件对象
if(window.event){
oEvent = window.event;
}
//获取事件源对象
var oTarget;
if(oEvent.srcElement){
oTarget = oEvent.srcElement;
}else{
oTarget = oEvent.target;
}
var className = oTarget.className;//获取当前被点击的li元素的class属性值
if(className==”num”){//如果是数字
resultDom.value += oTarget.innerHTML;//给result文本框赋值
num = oTarget.innerHTML;//给num变量赋值
}else if(className==”operator”){//如果是运算符
//当operator变量为空时，才能点击运算符
if(num!=null&&operator==null) {
resultDom.value += oTarget.innerHTML;//给result文本框赋值
operator = oTarget.innerHTML;//给operator运算符变量赋值
}
}else{
if(oTarget.innerHTML==”C”){//如果li内容为C,就是清空操作
resultDom.value = “”;//清空文本框
//恢复下面变量初始状态
num = null;
operator=null;
}else if(oTarget.innerHTML==”=”){//做运算操作
var result = eval(resultDom.value);//利用eval()方法，自动把字符串进行数学运算
resultDom.value = result;//把运算结果赋值回文本框
num=result;//把运算结果赋值给num变量，以便下次运算
operator=null;//把operator变量恢复初始状态
}
}
}
</script>

<script>
var ulDom2 = document.getElementById(“btnBox2”);
var liDoms2 = ulDom2.getElementsByTagName(“li”);
var resultDom2 = document.getElementById(“result2”);
var num1=null,num2=null;//定义保存第一个数字和第二个数字的变量
var operator=null;//定义保存运算符的变量

//给所有li元素绑定事件
for(var i=0;i<liDoms2.length;i++) {
liDoms2[i].onclick = function (oEvent) {
if (window.event) {
oEvent = window.event;
}
var oTarget;
if (oEvent.srcElement) {
oTarget = oEvent.srcElement;
} else {
oTarget = oEvent.target;
}

if(oTarget.className.indexOf(“num”)>=0){//点击数字li时
if(operator==null){//如果operator运算符为空，证明用户点击的是第一个数字
resultDom2.value += oTarget.innerHTML;//把li的文本累加到结果文本中
num1 = parseInt(resultDom2.value);//把最后的数字强制转化为整形后，保存到num1
}else{//如果operator运算符不为空，证明用户点击的是第二个数字
resultDom2.value += oTarget.innerHTML;//把li的文本累加到结果文本中
num2 = parseInt(resultDom2.value);//把最后的数字强制转化为整形后，保存到num1
}
}else if(oTarget.className.indexOf(“operator”)>=0){//点击操作符li时
//当第一个数字num1不为空时，才执行点击运算符的操作
if(num1!=null) {
operator = oTarget.innerHTML;//保存运算符
resultDom2.value = “”;//清空结果文本框
}
}else {
if (oTarget.innerHTML == “C”) {//点击清空C按钮li时
resultDom2.value = “”;//清空结果文本框
num1 = null;//清空num1
num2 = null;//清空num2
operator = null;//清空operator运算符
} else if (oTarget.innerHTML == “=”) {//点击等于号li时
//按照operator运算符进行相应的运算，把结果赋值到结果文本框中
switch (operator) {
case “+”:
resultDom2.value = num1 + num2;
break;
case “-“:
resultDom2.value = num1 - num2;
break;
case ““:
resultDom2.value = num1 num2;
break;
case “/“:
resultDom2.value = num1 / num2;
break;
default:
resultDom2.value = “出错了！”;
}
num1 = parseFloat(resultDom2.value);//最后把运算的结果赋值给num1，可以继续进行下一次的运算
num2 = null;//清空num2
operator = null;//清空operator运算符
}
}
}
}
</script>
